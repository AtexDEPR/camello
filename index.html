// Conectando talento con oportunidades
import { Freelancer, Client, Project } from '@camello/core';

const matchTalentWithOpportunity = async (freelancer: Freelancer, client: Client) => {
  const projects = await Project.findMatching({
    skills: freelancer.skills,
    availability: freelancer.availability,
    requirements: client.projectRequirements
  });
  
  return {
    matches: projects.filter(p => p.compatibilityScore > 0.8),
    topMatch: projects.sort((a, b) => b.compatibilityScore - a.compatibilityScore)[0]
  };
};

Sobre Nosotros

¿Qué es Camello?

Camello es una plataforma digital que conecta a freelancers con clientes en Colombia. Su objetivo es facilitar la contratación de profesionales independientes en diversas áreas, ofreciendo herramientas innovadoras para mejorar la experiencia de trabajo remoto.

Conectamos Talento

Unimos a profesionales independientes con empresas y clientes que necesitan sus servicios.

Sistema de Mentoría

Conectamos a nuevos freelancers con mentores experimentados que ganan reputación y comisiones.

Gamificación

Niveles y logros para recompensar a los freelancers y desbloquear beneficios exclusivos.

Asesoría Legal y Fiscal

Herramientas para calcular impuestos, generar facturas y consultorías legales en línea.

Nuestra Misión

Transformando el mercado freelance colombiano

Transformar el mercado freelance colombiano, creando un ecosistema donde profesionales independientes puedan desarrollar todo su potencial, acceder a oportunidades de calidad y construir carreras sostenibles, mientras las empresas encuentran el talento ideal para sus proyectos en un entorno seguro y eficiente.
